set lines 180
col path for a40
col name for a10
col HEADER_STATUS for a7
col FAILGROUP for a10
select d.name, d.path, d.HEADER_STATUS, d.failgroup, d.OS_MB
  from v$asm_disk d join v$asm_diskgroup g on (d.GROUP_NUMBER = g.GROUP_NUMBER)
 where g.NAME = 'OCR'
order by failgroup, name;
